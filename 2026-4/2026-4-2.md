### 状態管理
```
n = gets.to_i
s = gets.split.join  # ← ここがポイント

count = 0
state = 0  # 0:a待ち, 1:m待ち, 2:i待ち

s.each_char do |ch|
  if state == 0 && ch == "a"
    state = 1
  elsif state == 1 && ch == "m"
    state = 2
  elsif state == 2 && ch == "i"
    count += 1
    state = 0
  end
end

puts count
```
